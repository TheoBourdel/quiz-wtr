CREATE TABLE quiz (
    id SERIAL PRIMARY KEY,
    name VARCHAR(255) NOT NULL
);

CREATE TABLE question (
    id SERIAL PRIMARY KEY,
    quiz_id INT REFERENCES quiz(id),
    name TEXT NOT NULL
);

CREATE TABLE answer (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    is_correct BOOLEAN DEFAULT FALSE,
    question_id INT REFERENCES question(id)
);
